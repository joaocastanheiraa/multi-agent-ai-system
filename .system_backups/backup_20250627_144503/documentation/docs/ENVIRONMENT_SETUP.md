## 8. **Valida√ß√£o de Configura√ß√£o**

### 8.1. **Script de Valida√ß√£o**

```python
#!/usr/bin/env python3
# scripts/validate_env.py

import os
import sys
from typing import List, Tuple

def check_required_vars() -> List[Tuple[str, bool]]:
    """Verifica vari√°veis obrigat√≥rias"""
    required = [
        'OPENAI_API_KEY',
        'MCP_SERVER_PORT',
        'ENV'
    ]
    
    results = []
    for var in required:
        value = os.getenv(var)
        results.append((var, bool(value)))
    
    return results

def check_optional_vars() -> List[Tuple[str, bool]]:
    """Verifica vari√°veis opcionais"""
    optional = [
        'ANTHROPIC_API_KEY',
        'SUPABASE_URL',
        'REDIS_URL',
        'DATABASE_URL'
    ]
    
    results = []
    for var in optional:
        value = os.getenv(var)
        results.append((var, bool(value)))
    
    return results

def validate_api_keys():
    """Valida formato das API keys"""
    openai_key = os.getenv('OPENAI_API_KEY')
    if openai_key and not openai_key.startswith('sk-'):
        print(f"‚ö†Ô∏è  OPENAI_API_KEY parece inv√°lida")
    
    anthropic_key = os.getenv('ANTHROPIC_API_KEY')
    if anthropic_key and not anthropic_key.startswith('sk-ant-'):
        print(f"‚ö†Ô∏è  ANTHROPIC_API_KEY parece inv√°lida")

def main():
    print("üîç Validando configura√ß√£o de ambiente...\n")
    
    # Vari√°veis obrigat√≥rias
    print("üìã Vari√°veis Obrigat√≥rias:")
    required_results = check_required_vars()
    for var, exists in required_results:
        status = "‚úÖ" if exists else "‚ùå"
        print(f"  {status} {var}")
    
    # Vari√°veis opcionais
    print("\nüìã Vari√°veis Opcionais:")
    optional_results = check_optional_vars()
    for var, exists in optional_results:
        status = "‚úÖ" if exists else "‚ö†Ô∏è "
        print(f"  {status} {var}")
    
    # Valida√ß√£o adicional
    print("\nüîê Valida√ß√£o de API Keys:")
    validate_api_keys()
    
    # Resultado final
    missing_required = [var for var, exists in required_results if not exists]
    if missing_required:
        print(f"\n‚ùå Vari√°veis obrigat√≥rias faltando: {', '.join(missing_required)}")
        sys.exit(1)
    else:
        print(f"\n‚úÖ Configura√ß√£o b√°sica v√°lida!")

if __name__ == "__main__":
    main()
```
